1. Складіть запит для отримання усіх користувачів, відсортовані по імені
SELECT * FROM users
ORDER BY name;

2. Складіть запит для отримання усіх користувачів, вік яких більше 40 років
SELECT * FROM `users` WHERE age > 40;

3. Складіть запит для отримання усіх користувачів, вік яких від 20 до 40 років
SELECT * FROM `users` WHERE age > 20 AND age < 40;

SELECT * FROM `users` WHERE age BETWEEN 20 AND 40;

4. Складіть запит для отримання усіх користувачів, на рахунку яких 
більше 1000 (незалежно від валюти), до результату додати цей баланс
SELECT * FROM `users` JOIN accounts ON users.id=accounts.user_id WHERE balance > 1000;

SELECT users.name, SUM(accounts.balance) FROM `users` JOIN accounts ON users.id=accounts.user_id GROUP BY users.name HAVING SUM(accounts.balance) > 1000;

5. Складіть запит для отримання історії транзакцій, згрупувати по терміналам та вивести 
загальну суму, зняту з кожного терміналу, додати модель терміналу
SELECT SUM(amount) AS sumAmount, cashboxes.model
FROM logs
JOIN cashboxes
ON cashboxes.id=logs.cashbox_id
GROUP BY cashbox_id;

6. Складіть запит для отримання суми на балансі кожного терміналу (загальна сума коштів у терміналі)
SELECT SUM(amounts.value*amounts.quantly) AS allSum, cashbox_id
FROM amounts
GROUP BY cashbox_id;
